906typename CircularLinkedList<T>::iterator
907CircularLinkedList<T>::insert(iterator pos, const T& value) {
908 if (empty()) {
909 push_back(value);
910 return iterator(head, head);
911 }
912 if (pos == end()) {
913 push_back(value);
914 return iterator(head->prev, head);
915 }
916 Node* newNode = new Node(value);
917 Node* current = pos.current;
918 newNode->next = current;
919 newNode->prev = current->prev;
920 current->prev->next = newNode;
921 current->prev = newNode;
922 if (current == head) {
923 head = newNode;
924 }
925 _size++;
926 return iterator(newNode, head);
927}

930template<typename T>
931typename CircularLinkedList<T>::iterator
932CircularLinkedList<T>::erase(iterator pos) {
933 if (empty()) throw std::out_of_range("CircularLinkedList::erase: empty list");
934 if (pos == end()) throw std::invalid_argument("CircularLinkedList::erase: cannot erase end iterator");
935 Node* toDelete = pos.current;
936 Node* nextNode = toDelete->next;
937 if (_size == 1) {
938 delete toDelete;
939 head = nullptr;
940 _size = 0;
941 return end();

943 toDelete->prev->next = toDelete->next;
944 toDelete->next->prev = toDelete->prev;
945 if (toDelete == head) {

949 _size--;
